# i18n/l10n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make Quorant's backend i18n/l10n-capable by replacing hardcoded English error messages with structured message keys, adding locale/currency/timezone to org and financial models, and serving language packs via a public API endpoint.

**Architecture:** Error types gain `MessageKey` + `Params` fields replacing `Message`. A new `platform/i18n` package embeds JSON language packs and serves them via `GET /api/v1/i18n/{locale}`. Organization model gets `locale`, `timezone`, `currency_code`, `country`. Financial tables get `currency_code`. Notification templates get `locale` for per-locale resolution.

**Tech Stack:** Go stdlib (`embed`, `encoding/json`, `net/http`), PostgreSQL ALTER TABLE migrations, existing `api`/`middleware` packages. No new external dependencies.

**Spec:** `docs/superpowers/specs/2026-04-09-i18n-l10n-design.md`

```go
// Package api provides standard HTTP error types, response envelope helpers,
// and cursor-based pagination utilities used by every API endpoint.
package api

import (
	"fmt"
	"net/http"
)

// Param is a key-value pair passed to error constructors for i18n interpolation.
type Param struct {
	Key   string
	Value any
}

// P creates a Param. Shorthand used at every error call site.
func P(key string, value any) Param {
	return Param{Key: key, Value: value}
}

// paramsToMap converts variadic Params to a map.
func paramsToMap(params []Param) map[string]any {
	if len(params) == 0 {
		return nil
	}
	m := make(map[string]any, len(params))
	for _, p := range params {
		m[p.Key] = p.Value
	}
	return m
}

// APIError is implemented by all API-level errors. It exposes the HTTP status
// code, a machine-readable code string, and i18n message key + params.
type APIError interface {
	error
	StatusCode() int
	ErrorCode() string
	MsgKey() string
	MsgParams() map[string]any
}

// ValidationError represents a 400 Bad Request caused by invalid input data.
type ValidationError struct {
	messageKey string
	Field      string
	params     map[string]any
}

func NewValidationError(messageKey, field string, params ...Param) *ValidationError {
	return &ValidationError{messageKey: messageKey, Field: field, params: paramsToMap(params)}
}

func (e *ValidationError) Error() string {
	return fmt.Sprintf("validation error: %s (field=%s)", e.messageKey, e.Field)
}
func (e *ValidationError) StatusCode() int        { return http.StatusBadRequest }
func (e *ValidationError) ErrorCode() string      { return "VALIDATION_ERROR" }
func (e *ValidationError) MsgKey() string          { return e.messageKey }
func (e *ValidationError) MsgParams() map[string]any { return e.params }
```

```go
// Package i18n provides locale pack loading and serving for client-side translation.
package i18n

import (
	"embed"
	"encoding/json"
	"fmt"
	"path/filepath"
	"sort"
	"strings"
)

//go:embed locales/*.json
var localeFS embed.FS

// Pack is a single language pack served to clients.
type Pack struct {
	Locale   string            `json:"locale"`
	Version  string            `json:"version"`
	Messages map[string]string `json:"messages"`
}

// Registry holds all loaded language packs, keyed by locale string.
type Registry struct {
	packs map[string]*Pack
}

// NewRegistry loads all embedded locale JSON files and returns a Registry.
func NewRegistry() (*Registry, error) {
	entries, err := localeFS.ReadDir("locales")
	if err != nil {
		return nil, fmt.Errorf("i18n: read locales dir: %w", err)
	}

	packs := make(map[string]*Pack, len(entries))
	for _, entry := range entries {
		if entry.IsDir() || filepath.Ext(entry.Name()) != ".json" {
			continue
		}

		data, err := localeFS.ReadFile("locales/" + entry.Name())
		if err != nil {
			return nil, fmt.Errorf("i18n: read %s: %w", entry.Name(), err)
		}

		var pack Pack
		if err := json.Unmarshal(data, &pack); err != nil {
			return nil, fmt.Errorf("i18n: parse %s: %w", entry.Name(), err)
		}

		locale := strings.TrimSuffix(entry.Name(), ".json")
		pack.Locale = locale
		packs[locale] = &pack
	}

	return &Registry{packs: packs}, nil
}

// Get returns the pack for the given locale, or false if not found.
func (r *Registry) Get(locale string) (*Pack, bool) {
	p, ok := r.packs[locale]
	return p, ok
}

// Locales returns a sorted list of available locale codes.
func (r *Registry) Locales() []string {
	locales := make([]string, 0, len(r.packs))
	for k := range r.packs {
		locales = append(locales, k)
	}
	sort.Strings(locales)
	return locales
}
```

```go
package i18n

import (
	"encoding/json"
	"net/http"
)

// Handler serves language packs via HTTP.
type Handler struct {
	registry *Registry
}

// NewHandler creates a Handler backed by the given Registry.
func NewHandler(registry *Registry) *Handler {
	return &Handler{registry: registry}
}

// GetPack serves GET /api/v1/i18n/{locale}.
func (h *Handler) GetPack(w http.ResponseWriter, r *http.Request) {
	locale := r.PathValue("locale")

	pack, ok := h.registry.Get(locale)
	if !ok {
		http.NotFound(w, r)
		return
	}

	w.Header().Set("Content-Type", "application/json")
	w.Header().Set("Cache-Control", "public, max-age=3600")
	w.Header().Set("ETag", `"`+pack.Version+`"`)
	_ = json.NewEncoder(w).Encode(pack)
}

// ListLocales serves GET /api/v1/i18n.
func (h *Handler) ListLocales(w http.ResponseWriter, r *http.Request) {
	w.Header().Set("Content-Type", "application/json")
	_ = json.NewEncoder(w).Encode(map[string]any{
		"locales": h.registry.Locales(),
	})
}
```
